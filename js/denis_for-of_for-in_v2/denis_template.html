Boucles for(in) et for(of)

par Frederick Denis

                
                    
function bioActeurForOf(){

    const acteur = ["Chuck", "Norris", "'77"];

    for (let item of acteur){
        document.querySelector("#sectionaffichageforof").innerHTML += item +" ";
    }
}

For (of) en action !

                
                    
function bioActeurForIn(){

    const acteur = {prenom:"Chuck", nom:"Norris", age:77};

    for (let prop in acteur){
        document.querySelector("#sectionaffichageforin").innerHTML += acteur[prop];
    }

}

For (in) en action !
