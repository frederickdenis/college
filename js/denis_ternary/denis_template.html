function nombrePlusGrand(a, b, c){

    return ((a > b) && (a > c)) ? a : (b > c) ? b : c;

}

let resultat = nombrePlusGrand(1, 2, 3);

console.log(resultat);
document.querySelector("#affichage").innerHTML = "Le nombre le plus grand est " + resultat + " ."
